    ```python
    def catch_and_log(func):
        async def wrapper(*args, **kwargs):
            try:
                return await func(*args, **kwargs)
            except Exception as e:
                # LOG IMMEDIATE ERROR WITH CONTEXT
                logger.error(f"‚ùå ERROR in [{func.__name__}]")

        return wrapper
    ```
